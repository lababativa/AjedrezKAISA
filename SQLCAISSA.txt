SQL CAISSA


create table persona (
	idPersona int NOT NULL AUTO_INCREMENT,
	PrimerNombre varchar(20) ,
	SegundoNombre varchar(20) ,
	Edad integer ,
	Correo varchar(40) ,
	PRIMARY KEY(idPersona)
);

create table partida (
	idPartida int NOT NULL AUTO_INCREMENT ,
	idPersona int , 
	idNivelModalidad int,
	TiempoTotal time,
	JugadasRealizadas int,
	CantidadErrores int,
	CantidadPausas int,
	CantidadRetiradas int,
	TurnosDevueltos  int,
	PRIMARY KEY(idPartida)
);

create table nivelModalidad(
	idNivel int NOT NULL AUTO_INCREMENT,
	idNivelModalidad int ,
	PRIMARY KEY(idNivel)
);

create table modalidad(
	idModalidad int NOT NULL AUTO_INCREMENT,
	Nombre varchar(30),
	Descripcion varchar(200),
	Documentacion text,
	PRIMARY KEY(IDModalidad)
);

create table informe (
	idInforme int NOT NULL AUTO_INCREMENT,
	idPartida int,
	Fecha DATE,
	PorcentajeError int,
	ContrasteJugadas int,
	MovimientoInvalido int, 
	PRIMARY KEY(idInforme)
);

create table informeGeneral(
	idInformeGeneral int NOT NULL AUTO_INCREMENT,
	NumeroPartidas int,
	PromedioError double,
	PromedioTiempo time,
	idPersona int,
	PromedioDevoluciones double,
	TotalMovimientosInvalidos int,
	PromedioRetirada double,
	PRIMARY KEY (idInformeGeneral)
	
);
create table nivel (
	idNivel int NOT NULL AUTO_INCREMENT,
	NumeroJugadas int,
	ReinaPosicion varchar(3),
	ReyPosicion varchar(3),
	Torre1Posicion varchar(3),
	Torre2Posicion varchar(3),
	Caballo1Posicion varchar(3),
	Caballo2Posicion varchar(3),
	Peon1Posicion varchar(3),
	Peon2Posicion varchar(3),
	Peon3Posicion varchar(3),
	Peon4Posicion varchar(3),
	Peon5Posicion varchar(3),
	Peon6Posicion varchar(3),
	Peon7Posicion varchar(3),
	Peon8Posicion varchar(3),
	Tiempo TIME,
	PRIMARY KEY (idNivel)
);
